    const c = document.getElementById('game');
    const g = c.getContext('2d');
    const tile = 24, cols = c.width / tile, rows = c.height / tile;
    let snake, dir, food, speed, alive;

    function reset() {
      snake = [{x: 8, y: 10}];
      dir = {x: 1, y: 0};
      food = spawnFood();
      speed = 120;
      alive = true;
    }
    function spawnFood() {
      let f;
      do {
        f = { x: (Math.random()*cols)|0, y: (Math.random()*rows)|0 };
      } while (snake.some(s => s.x===f.x && s.y===f.y));
      return f;
    }
    function step() {
      if (!alive) return;
      const head = { x: (snake[0].x + dir.x + cols) % cols,
                     y: (snake[0].y + dir.y + rows) % rows };
      if (snake.some(s => s.x===head.x && s.y===head.y)) { alive=false; draw(); return; }
      snake.unshift(head);
      if (head.x===food.x && head.y===food.y) {
        food = spawnFood(); speed = Math.max(60, speed-3);
      } else snake.pop();
      draw();
      setTimeout(step, speed);
    }
    function draw() {
      g.fillStyle = '#0d131c'; g.fillRect(0,0,c.width,c.height);
      // grid
      g.strokeStyle = '#121b27'; g.lineWidth = 1;
      for (let i=1;i<cols;i++) { g.beginPath(); g.moveTo(i*tile,0); g.lineTo(i*tile,c.height); g.stroke(); }
      for (let j=1;j<rows;j++) { g.beginPath(); g.moveTo(0,j*tile); g.lineTo(c.width,j*tile); g.stroke(); }

      g.fillStyle = '#ff6b6b'; g.fillRect(food.x*tile+4, food.y*tile+4, tile-8, tile-8);
      // snake

        g.fillStyle = i? '#5ea1ff' : '#8bd6ff';
        g.fillRect(s.x*tile+2, s.y*tile+2, tile-4, tile-4);
      });
      if (!alive) {
